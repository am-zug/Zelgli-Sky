function buildSVG() { svg.innerHTML = ""; const scene = document.createElementNS(SVG_NS, "g"); const lightH = 78; const normalW = 58; const wideW = normalW * 1.35; const gapX = 7; const largeDarkH = lightH * 2.5 * 0.6; const smallDarkH = lightH * 1.5 * 0.6; const baseY = 330; const numberY = 410; let x = 0; [...APARTMENTS].reverse().forEach(apt => { const isSmall = [4, 5, 11, 12].includes(apt.id); const isWide = [4, 5].includes(apt.id); const unitW = isWide ? wideW : normalW; const darkH = isSmall ? smallDarkH : largeDarkH; const colors = getStatusColors(apt.status); // 180° + horizontal mirror = vertical flip // light block on top, dark block below const balconyY = baseY - lightH - darkH; const livingY = baseY - darkH; const unit = document.createElementNS(SVG_NS, "g"); unit.setAttribute("class", "apt-unit"); unit.setAttribute("id", `svg-apt-${apt.id}`); unit.setAttribute("data-id", apt.id); unit.appendChild(createRect(x, balconyY, unitW, lightH, colors.balcony, "balcony")); unit.appendChild(createRect(x, livingY, unitW, darkH, colors.living, "living")); unit.appendChild(createText(x + unitW / 2, numberY, apt.id, 18)); unit.addEventListener("click", () => { if (apt.factsheet) window.open(apt.factsheet, "_blank", "noopener"); }); scene.appendChild(unit); x += unitW + gapX; }); svg.appendChild(scene); const bbox = scene.getBBox(); const vb = svg.viewBox.baseVal; const margin = 20; const scale = Math.min( (vb.width - margin * 2) / bbox.width, (vb.height - margin * 2) / bbox.height ); const tx = (vb.width - bbox.width * scale) / 2 - bbox.x * scale; const ty = (vb.height - bbox.height * scale) / 2 - bbox.y * scale; scene.setAttribute("transform", `translate(${tx}, ${ty}) scale(${scale})`); svg.querySelectorAll(".apt-unit").forEach(el => { el.addEventListener("mouseenter", () => highlightApt(el.dataset.id, true)); el.addEventListener("mouseleave", () => highlightApt(el.dataset.id, false)); }); }
